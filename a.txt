def f(a):
    # 函数开始
	shuzi=0
	daxie=0
	xiaoxie=0
	for i in range(len(a)):
		if ord(a[i])>=65 and ord(a[i])<=90:
			daxie += 1
		elif ord(a[i])>=97 and ord(a[i])<=122:
			xiaoxie += 1
		elif ord(a[i])>=48 and ord(a[i])<=57:
			shuzi += 1
	return [daxie, xiaoxie, shuzi]
    # 函数结束
print(f('12ABDC') == [4,0,2])
print(f('abcAAAABC') == [6,3,0])
print(f('abccc1ss12ABC') == [3,7,3])
print(f('DFK@!!dfjso16  52 @# 132') == [3,5,7])
